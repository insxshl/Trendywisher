    <script>

        const wishLib = {

            "Birthday": "May your digital soul dance in the neon lights of another great year. Happy Birthday!",

            "New Year": "System Update: Entering a new year. May your success reach maximum capacity!",

            "Marriage": "Two systems, one core. Wishing you a lifetime of synchronized happiness!"

        };

        function handleGenerate() {

            const s = document.getElementById('sender').value || "Friend";

            const r = document.getElementById('receiver').value || "You";

            const o = document.getElementById('occasion').value;

            const message = `Hey ${r}, ${wishLib[o]}`;

            showWish(message, s);

        }

        function showWish(msg, from) {

            document.getElementById('setup-view').classList.add('hidden');

            document.getElementById('output-view').classList.remove('hidden');

            document.getElementById('display-wish').innerText = msg;

            document.getElementById('display-from').innerText = `— Sent by: ${from}`;

            // Update URL for sharing

            const shareUrl = `${window.location.origin}${window.location.pathname}?m=${btoa(encodeURIComponent(msg))}&f=${btoa(encodeURIComponent(from))}`;

            window.history.pushState({path: shareUrl}, '', shareUrl);

           

            // Scroll to top to ensure card is seen

            window.scrollTo(0,0);

        }

        function copyToClipboard() {

            const text = `${document.getElementById('display-wish').innerText} ${document.getElementById('display-from').innerText}\nShared via: ${window.location.href}`;

            navigator.clipboard.writeText(text).then(() => {

                alert("Copied to Clipboard!");

            });

        }

        async function shareSystem() {

            const text = `${document.getElementById('display-wish').innerText} ${document.getElementById('display-from').innerText}`;

            const url = window.location.href;

            if (navigator.share) {

                try {

                    await navigator.share({

                        title: 'A Special Wish for You',

                        text: text,

                        url: url,

                    });

                } catch (err) {

                    // Fail silently or handle cancel

                }

            } else {

                // FALLBACK FOR DESKTOP: Copy link automatically

                copyToClipboard();

                alert("Share not supported on this browser. Link copied instead!");

            }

        }

        window.onload = () => {

            const params = new URLSearchParams(window.location.search);

            if (params.has('m') && params.has('f')) {

                try {

                    const msg = decodeURIComponent(atob(params.get('m')));

                    const from = decodeURIComponent(atob(params.get('f')));

                    showWish(msg, from);

                } catch(e) { console.error("Invalid URL params"); }

            }

        };

        // Keyboard awareness: Hide ad when keyboard opens

        window.visualViewport.addEventListener('resize', () => {

            const ad = document.getElementById('sticky-ad');

            ad.style.display = window.visualViewport.height < window.innerHeight * 0.8 ? 'none' : 'flex';

        });
